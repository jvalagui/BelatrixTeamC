git checkout -b <rama>  	>> 	posicionarse en la rama (permite navegar entre ramas)
git commit 					>> 	guardar cambios
git merge <rama>		 	>> 	para integrar <rama> a la rama actual
git status 					>> 	verificar si todo esta bien
git push origin master 		>> 	para subir el ultimo commit de la rama actual
git pull origin master 		>> 	para jalar el ultimo commit de la rama actual
git reset --hard 			>>	para posicionarse en el ultimo commit, (--hard : perdiendo todos los cambios locales)
git log 					>> 	para ver el historial de commits
git add <nombreArchivo>     >> 	para agregar un archivo al seguimiento de control
git commit -m <mensaje>		>> 	mensaje de commit
